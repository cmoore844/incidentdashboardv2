# ##Fetch image from Dockerhub
# FROM node:latest

# #Create the directory within the container
# RUN mkdir -p /app

# #Copy source files from local repo to the container directory
# COPY ./ /app

# #PWD of the container
# WORKDIR /app

# #Install npm packages
# RUN npm install 

# #Run the application exposing port 3000
# EXPOSE 3000

# #Start the application
# CMD ["npm", "run", "dev"]
